    const grid = document.getElementById('productGrid');
    const searchInput = document.getElementById('searchInput');
    const noResults = document.getElementById('no-results');
    let cartCounter = 0;

    // 2. Funci√≥n para mostrar productos
    function displayProducts(filteredList) {
        grid.innerHTML = '';
        if (filteredList.length === 0) {
            noResults.style.display = 'block';
        } else {
            noResults.style.display = 'none';
            filteredList.forEach(product => {
                grid.innerHTML += `
                    <div class="card">
                        <img src="${product.img}" alt="${product.name}">
                        <div class="card-body">
                            <span class="card-category">${product.category}</span>
                            <div class="card-title">${product.name}</div>
                            <div class="price">$${product.price.toLocaleString('es-CL')}</div>
                            <button class="btn-add" onclick="addToCart()">Agregar al carro</button>
                        </div>
                    </div>
                `;
            });
        }
    }

    // 3. L√≥gica del Buscador
    searchInput.addEventListener('input', (e) => {
        const text = e.target.value.toLowerCase();
        const filtered = products.filter(p => p.name.toLowerCase().includes(text));
        displayProducts(filtered);
    });

    // 4. L√≥gica de Filtros por Categor√≠a
    function filterProducts(cat) {
        document.getElementById('category-title').innerText = cat === 'todos' ? 'Todos los productos' : cat;
        if (cat === 'todos') {
            displayProducts(products);
        } else {
            const filtered = products.filter(p => p.category === cat);
            displayProducts(filtered);
        }
    }

    // 5. Simular Carrito
    function addToCart() {
        cartCounter++;
        document.getElementById('cart-count').innerText = cartCounter;
    }

    // Inicializar la p√°gina
    displayProducts(products);
